```csharp
var options = new CsvOptions
{
    DataCategory = DataCategory.Resources,
    TextDelimiter = CsvTextDelimiter.Semicolon,
    Encoding = Encoding.Unicode,
    IncludeHeaders = true
};
```

project.Save(OutDir + "WorkWithCsvOptions_out.csv", options);
